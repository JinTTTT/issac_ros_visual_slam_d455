
>>> 1. Launch container
isaac_ros_container

>>> 2. Intall package's dependecy
sudo apt-get install -y ros-humble-isaac-ros-visual-slam

>>> 3. Launch isaac_ros_visual_slam
# only run vslam
ros2 launch isaac_ros_visual_slam isaac_ros_visual_slam.launch.py
# run both camera and vslam
ros2 launch isaac_ros_visual_slam isaac_ros_visual_slam_realsense.launch.py

>>> 4. run RVIZ2 display output
rviz2 -d src/isaac_ros_visual_slam/isaac_ros_visual_slam/rviz/default.cfg.rviz


>>> 5. Record a rosbag
ros2 bag record -o my_realsense_data \
  /camera/infra1/image_rect_raw \
  /camera/infra2/image_rect_raw \
  /camera/infra1/camera_info \
  /camera/infra2/camera_info \
  /camera/imu

>>> 1. if have GPG error while trying to update your package lists.

# --- Step 1: Install prerequisite tools ---
sudo apt-get update
sudo apt-get install -y \
    curl \
    wget \
    gpg

# --- Step 2: Configure the ROS 2 Repository ---

sudo curl -sSL https://raw.githubusercontent.com/ros/rosdistro/master/ros.key -o /usr/share/keyrings/ros-archive-keyring.gpg

# --- Step 3: Configure the Kitware (for CMake) Repository ---
wget -qO - https://apt.kitware.com/keys/kitware-archive-latest.asc | sudo gpg --dearmor -o /etc/apt/trusted.gpg.d/kitware.gpg

echo "deb [signed-by=/etc/apt/trusted.gpg.d/kitware.gpg] https://apt.kitware.com/ubuntu focal main" | sudo tee /etc/apt/sources.list.d/kitware.list > /dev/null

# --- Step 4: Final Update ---
sudo apt-get update
